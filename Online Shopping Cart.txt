import java.sql.*;
import java.util.Scanner;

public class ShoppingCart {

    static final String URL = "jdbc:mysql://localhost:3306/shopping_cart";
    static final String USER = "root";
    static final String PASS = "root";

    static Connection conn;

    public static void main(String[] args) {
        try {
            conn = DriverManager.getConnection(URL, USER, PASS);
            Scanner sc = new Scanner(System.in);

            while (true) {
                System.out.println("\n1. View Products");
                System.out.println("2. Add to Cart");
                System.out.println("3. View Cart");
                System.out.println("4. Checkout");
                System.out.println("5. Exit");
                System.out.print("Enter choice: ");

                int ch = sc.nextInt();

                switch (ch) {
                    case 1: viewProducts(); break;
                    case 2: addToCart(sc); break;
                    case 3: viewCart(); break;
                    case 4: checkout(); break;
                    case 5: System.exit(0);
                    default: System.out.println("Invalid choice");
                }
            }

        } catch (Exception e) {
            System.out.println(e);
        }
    }

    static void viewProducts() {
        try {
            Statement stmt = conn.createStatement();
            ResultSet rs = stmt.executeQuery("SELECT * FROM products");

            System.out.println("\nAvailable Products:");
            while (rs.next()) {
                System.out.println(rs.getInt(1) + ". " + rs.getString(2) + " - ₹" + rs.getDouble(3));
            }

        } catch (Exception e) {
            System.out.println(e);
        }
    }

    static void addToCart(Scanner sc) {
        try {
            System.out.print("Enter product ID: ");
            int id = sc.nextInt();

            PreparedStatement ps = conn.prepareStatement("SELECT * FROM products WHERE id=?");
            ps.setInt(1, id);
            ResultSet rs = ps.executeQuery();

            if (rs.next()) {
                String name = rs.getString("name");
                double price = rs.getDouble("price");

                PreparedStatement insert = conn.prepareStatement(
                        "INSERT INTO cart (product_name, price) VALUES (?, ?)");
                insert.setString(1, name);
                insert.setDouble(2, price);
                insert.executeUpdate();

                System.out.println("Added to cart");
            } else {
                System.out.println("Product not found");
            }

        } catch (Exception e) {
            System.out.println(e);
        }
    }

    static void viewCart() {
        try {
            Statement stmt = conn.createStatement();
            ResultSet rs = stmt.executeQuery("SELECT * FROM cart");

            double total = 0;
            System.out.println("\nCart Items:");

            while (rs.next()) {
                System.out.println(rs.getString("product_name") + " - ₹" + rs.getDouble("price"));
                total += rs.getDouble("price");
            }

            System.out.println("Total: ₹" + total);

        } catch (Exception e) {
            System.out.println(e);
        }
    }

    static void checkout() {
        try {
            Statement stmt = conn.createStatement();
            ResultSet rs = stmt.executeQuery("SELECT SUM(price) FROM cart");

            if (rs.next()) {
                double total = rs.getDouble(1);
                System.out.println("\nTotal amount: ₹" + total);
                System.out.println("Order placed successfully!");
            }

            stmt.executeUpdate("DELETE FROM cart");

        } catch (Exception e) {
            System.out.println(e);
        }
    }
}
